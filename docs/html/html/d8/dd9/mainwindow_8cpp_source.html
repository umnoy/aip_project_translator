162QString MainWindow::detectLanguage(const QString &text)
163{
164 if (text.isEmpty()) {
165 return "Неизвестный";
166 }
167
168 bool hasCyrillic = false;
169 bool hasLatin = false;
170
171 for (const QChar &c : text) {
172 if (c.unicode() >= 0x0400 && c.unicode() <= 0x04FF) { // Кириллица
173 hasCyrillic = true;
174 } else if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z')) { // Латиница
175 hasLatin = true;
176 }
177 }
178
179 if (hasCyrillic && !hasLatin) {
180 return "Русский";
181 } else if (hasLatin && !hasCyrillic) {
182 return "Английский";
183 } else {
184 return "Неизвестный";
185 }
186}
